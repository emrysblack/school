a. 

maxPos(list, l, r)
IF l == r
   RETURN l
SET mid to (l + r) / 2
SET left to maxPos(list,l,mid)
SET right to maxPos(list,mid+1,r)
IF list of left > list of right
   RETURN left
ELSE
   RETURN right
